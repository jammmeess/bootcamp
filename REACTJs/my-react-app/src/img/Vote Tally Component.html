The same goes for the Vice President table row.

  2. Inside the handleSubmit method, you need to update the state variable formSubmitted to true to indicate that a form submission has occurred. Add the following line of code inside the method:

  3. Modify the table row rendering for Mayor, Vice President, and Class Representative sections. Replace (props.mayor = 0 ? "gray" : "black") with (mayor.totalVotes === 0 ? "gray" : "black") and (props.classrep = 0 ? "gray" : "black") with (classrep.totalVotes === 0 ? "gray" : "black").

javascript
class Elections extends Component {
  constructor(props) {
    super(props);
    this.state = {
      forMayor: forMayor.map((candidate) => ({ ...candidate, totalVotes: 0 })),
      forVPres: forVPres.map((candidate) => ({ ...candidate, totalVotes: 0 })),
      forCRep: forCRep.map((candidate) => ({ ...candidate, totalVotes: 0 })),
      formSubmitted: false,
    };
  }

  voteMayor = (event) => {
    const selectedMayor = event.target.value;
    this.setState((prevState) => ({
      forMayor: prevState.forMayor.map((candidate) => {
        if (candidate.name === selectedMayor) {
          return { ...candidate, totalVotes: candidate.totalVotes + 1 };
        }
        return candidate;
      }),
    }));
  };

  voteVPres = (event) => {
    const selectedVPres = event.target.value;
    this.setState((prevState) => ({
      forVPres: prevState.forVPres.map((candidate) => {
        if (candidate.name === selectedVPres) {
          return { ...candidate, totalVotes: candidate.totalVotes + 1 };
        }
        return candidate;
      }),
    }));
  };

  voteCRep = (event) => {
    const selectedCRrep = event.target.value;
    this.setState((prevState) => ({
      forCRep: prevState.forCRep.map((candidate) => {
        if (candidate.name === selectedCRrep) {
          return { ...candidate, totalVotes: candidate.totalVotes + 1 };
        }
        return candidate;
      }),
    }));
  };

  handleSubmit = (event) => {
    event.preventDefault();
    this.setState({ formSubmitted: true });
  };

  render() {
    return (
      <>
        <h1>Voting form</h1>
        <div>
          <form onSubmit={this.handleSubmit}>
            {/* Form content */}
          </form>
        </div>
        <div>
          <table className="table">
            <thead>
              <tr>
                <th>Name</th>
                <th>Position</th>
                <th>Total Votes</th>
              </tr>
            </thead>
            <tbody>
              {this.state.forMayor.map((mayor) => (
                <tr
                  key={mayor.name}
                  className={mayor.totalVotes === 0 ? "gray" : "black"}
                >
                  <td>{mayor.name}</td>
                  <td>Mayor</td>
                  <td>{mayor.totalVotes}</td>
                </tr>
              ))}
              {this.state.forVPres.map((vicepres) => (
                <tr
                  key={vicepres.name}
                  className={vicepres.totalVotes === 0 ? "gray" : "black"}
                >
                  <td>{vicepres.name}</td>
                  <td>Vice President</td>
                  <td>{vicepres.totalVotes}</td>
                </tr>
              ))}
              {this.state.forCRep.map((classrep) => (
                <tr
                  key={classrep.name}
                  className={classrep.totalVotes === 0 ? "gray" : "black"}
                >
                  <td>{classrep.name}</td>
                  <td>Class Representative</td>
                  <td>{classrep.totalVotes}</td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </>
    );
  }
}

These modifications ensure that the vote counting and row coloration work as expected.
